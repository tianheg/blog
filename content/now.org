<script>
document.addEventListener('DOMContentLoaded', () => {
  const container = document.getElementById('task-list');
  
  fetch('https://tianheg.co/api/fogg/list')
    .then(response => {
      if (!response.ok) {
        throw new Error(`HTTP error! status: ${response.status}`);
      }
      return response.json();
    })
    .then(result => {
      if (result.status !== "ok") {
        throw new Error(result.message || 'API response error');
      }
      
      // Reverse the array before processing
      result.data.reverse().forEach(task => {
      const taskElement = document.createElement('div');
      taskElement.className = `task-item ${task.completed ? 'completed' : ''}`;
      
      const content = document.createElement('span');
      content.innerHTML = `[${task.date}]${task.title}`;
      
      taskElement.appendChild(content);
      container.appendChild(taskElement);
      });
    })
    .catch(error => {
      console.error('Error:', error);
      container.innerHTML = `<div class="error">Error loading tasks: ${error.message}</div>`;
    });
});
</script>
