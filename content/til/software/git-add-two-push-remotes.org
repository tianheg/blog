#origin git@github.com:xxx/a.git (fetch)
#origin git@codeberg.org:xxx/a.git (push)

#origin git@github.com:xxx/a.git (fetch)
#origin git@codeberg.org:xxx/a.git (push)
#origin git@codeberg.org:xxx/b.git (push)
