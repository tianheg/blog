{{- /* Custom reading time and word count for mixed CJK/English content. Strips Org code blocks (#+BEGIN_SRC ... #+END_SRC) and frontmatter. Word count: CJK chars (matched via \p{Han}) = 1 word each, remaining whitespace-separated tokens = 1 word each. Reading speed: ~200 wpm (mixed CJK/English). */ -}} {{- $raw := .RawContent -}} {{- /* Strip Org code blocks */ -}} {{- $noCode := "" -}} {{- $inCode := false -}} {{- range (split $raw "\n") -}} {{- if or (hasPrefix . "#+BEGIN_SRC") (hasPrefix . "```") -}} {{- $inCode = true -}} {{- else if or (hasPrefix . "#+END_SRC") (hasPrefix . "```") -}} {{- $inCode = false -}} {{- else if not $inCode -}} {{- $noCode = print $noCode . " " -}} {{- end -}} {{- end -}} {{- /* Strip Org frontmatter (#+ lines) */ -}} {{- $clean := "" -}} {{- range (split $noCode "\n") -}} {{- if not (hasPrefix . "#+") -}} {{- $clean = print $clean . " " -}} {{- end -}} {{- end -}} {{- /* CJK words: each Han character = 1 word */ -}} {{- $cjkMatches := findRE "\\p{Han}" $clean -}} {{- $cjkWords := len $cjkMatches -}} {{- /* Non-CJK words: strip Han chars, split whitespace, count tokens */ -}} {{- $asciiOnly := replaceRE "\\p{Han}" " " $clean -}} {{- $asciiWords := 0 -}} {{- range split $asciiOnly " " -}} {{- $token := trim . " \t\n\r,.!?;:\"'()[]{}…—-–" -}} {{- if gt (len $token) 0 -}} {{- $asciiWords = add $asciiWords 1 -}} {{- end -}} {{- end -}} {{- $words := add $cjkWords $asciiWords -}} {{- if lt $words 3 -}}{{- $words = 3 -}}{{- end -}} {{- /* Reading time: ~200 wpm for mixed content */ -}} {{- $mins := div (add $words 199) 200 -}} {{- if lt $mins 1 -}}{{- $mins = 1 -}}{{- end -}} {{ $mins }} min read · {{ $words }} words
