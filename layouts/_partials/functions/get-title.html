{{/* Get page title with unified fallback chain: .Title → first Org-mode heading (* ) → .Params.title → fallback (default: BaseFileName) Usage: {{ partial "functions/get-title" (dict "page" .) }} {{ partial "functions/get-title" (dict "page" . "fallback" .Site.Title) }} */}} {{- $page := .page -}} {{- $fallback := .fallback -}} {{- $title := $page.Title -}} {{- if not $title -}} {{- with $page.RawContent -}} {{- $lines := split . "\n" -}} {{- range $lines -}} {{- if and (not $title) (hasPrefix . "* ") -}} {{- $title = trim (strings.TrimPrefix "* " .) " " -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- if not $title -}} {{- $title = $page.Params.title -}} {{- end -}} {{- if not $title -}} {{- $title = $fallback | default $page.File.BaseFileName -}} {{- end -}} {{- return $title -}}
