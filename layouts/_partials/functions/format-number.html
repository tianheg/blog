{{/* Format a number with thousand separators */}} {{- $n := . -}} {{- $s := printf "%.0f" (float $n) -}} {{- $result := "" -}} {{- $len := len $s -}} {{- range $i, $c := seq $len -}} {{- $pos := sub $len $i -}} {{- $digit := substr $s (sub $pos 1) 1 -}} {{- $result = printf "%s%s" $digit $result -}} {{- if and (lt $i (sub $len 1)) (eq (mod $i 3) 0) -}} {{- $result = printf ",%s" $result -}} {{- end -}} {{- end -}} {{- $result -}}
