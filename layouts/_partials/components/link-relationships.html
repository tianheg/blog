{{/* ============================================ Parse Org-mode links and build page relationships ============================================ */}} {{/* Extract all org-mode format links [[path][title]] from raw content */}} {{- $rawContent := .RawContent -}} {{- $orgLinkPattern := `\[\[([^\]]+)\]\[[^\]]*\]\]` -}} {{- $matches := findRE $orgLinkPattern $rawContent -}} {{- range $matches -}} {{/* Parse link format and extract the path portion */}} {{- $inner := strings.TrimPrefix "[[" . -}} {{- $inner := strings.TrimSuffix "]]" $inner -}} {{- $parts := split $inner "][" -}} {{- $linkPath := index $parts 0 -}} {{/* Skip external links (starting with http/https) */}} {{- if not (hasPrefix $linkPath "http") -}} {{/* Clean path: remove leading slashes */}} {{- $cleanPath := $linkPath -}} {{- if hasPrefix $linkPath "/" -}} {{- $cleanPath = strings.TrimPrefix "/" $linkPath -}} {{- end -}} {{/* Only search for target pages in allowed sections */}} {{- $target := "" -}} {{- $allowedSections := slice "posts" "til" -}} {{/* Method 1: Direct path lookup */}} {{- range $allowedSections -}} {{- if not $target -}} {{- $tryPath := printf "%s/%s" . $cleanPath -}} {{- $target = site.GetPage $tryPath -}} {{- end -}} {{- end -}} {{/* Method 2: Match by filename or permalink */}} {{- if not $target -}} {{- range where site.RegularPages "Section" "in" $allowedSections -}} {{- if not $target -}} {{- if or (eq .File.BaseFileName $cleanPath) (hasSuffix .RelPermalink $cleanPath) -}} {{- $target = . -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{/* After finding the target, establish bidirectional link relationships */}} {{- if $target -}} {{/* Add "incoming" link to target page (current page links to it) */}} {{- with $target.Scratch.Get "incoming" -}} {{- $target.Scratch.Add "incoming" $.Page -}} {{- else -}} {{- $target.Scratch.Set "incoming" (slice $.Page) -}} {{- end -}} {{/* Add "outgoing" link from current page (links to target page) */}} {{- with $.Page.Scratch.Get "outgoing" -}} {{- $.Page.Scratch.Add "outgoing" $target -}} {{- else -}} {{- $.Page.Scratch.Set "outgoing" (slice $target) -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}}
